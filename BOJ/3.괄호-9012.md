```python
import sys
N = int(input())

for _ in range(N):
  
  command = list(sys.stdin.readline())
  command.pop() # \n 삭제

  stack = []
  for i in range(0, len(command)):
    if len(stack) == 0:
      stack.append(command[i])
    else:
      if command[i] == '(':
        stack.append(command[i])
      elif command[i] == ')':
        if len(stack) > 0 and stack[-1] == '(':
          stack.pop()
        else:
          stack.append(')')
          break
 
  if len(stack) > 0:
    print("NO")
  else:
    print("YES")
  ```
