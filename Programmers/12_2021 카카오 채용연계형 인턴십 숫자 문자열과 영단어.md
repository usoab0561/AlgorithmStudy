def solution(s):
    if 'one' in s:
        s= s.replace('one', '1')
    if 'zero' in s:
        s= s.replace('zero', '0')
    if 'two' in s:
        s= s.replace('two', '2')
    if 'three' in s:
        s= s.replace('three', '3')
    if 'four' in s:
        s= s.replace('four', '4')
    if 'five' in s:
        s= s.replace('five', '5')
    if 'six' in s:
        s= s.replace('six', '6')
    if 'seven' in s:
        s= s.replace('seven', '7')
    if 'eight' in s:
        s= s.replace('eight', '8')
    if 'nine' in s:
        s= s.replace('nine', '9')

    return int(s)

def solution(s):

    new_list = []

    # for i in range(len(s)):
    while i < len(s):
        if s[i].isalpha() == True:
            
            if s[i] == 'o':
                new_list.append(1)
                i += 2
            elif s[i] == 'z':
                new_list.append(0)
                i += 3
            elif s[i] == 't' and s[i+1] == 'w':
                new_list.append(2)
                i += 2
            elif s[i] == 't' and s[i+1] == 'h':
                new_list.append(3)
                i += 4
            elif s[i] == 'f' and s[i+1] == 'o':
                new_list.append(4)
                i += 3
            elif s[i] == 'f' and s[i+1] == 'i':
                new_list.append(5)
                i += 3
            elif s[i] == 's' and s[i+1] == 'i':
                new_list.append(6)
                i += 2

                new_list.append(7)
                i += 4
            elif s[i] == 'e':
                new_list.append(8)
                i += 4
            else:
                new_list.append(9)

            new_list.append(int(s[i]))
